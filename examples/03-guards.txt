-- Guards (Conditional Logic)
-- Guards with otherwise
absolute n
  | n < 0     = -n
  | otherwise = n

-- Multiple guard conditions
classify n
  | n < 0     = "negative"
  | n == 0    = "zero"
  | n > 0     = "positive"

-- Guards with pattern matching
sign 0 = 0
sign n
  | n > 0     = 1
  | otherwise = -1

-- absolute (-5)     → 5
-- classify (-3)     → "negative"
-- sign 42           → 1
